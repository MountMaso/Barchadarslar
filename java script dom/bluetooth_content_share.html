let defaultSize=document.getElementById('default');
let decreaseSize=document.getElementById('decrease');
let increaseSize=document.getElementById('increase');
let box=document.getElementById('box');
defaultSize.addEventListener('click',defaultSizeF);
decreaseSize.addEventListener('click',decreaseSizeF);
increaseSize.addEventListener('click',increaseSizeF);

let size=16;
function defaultSizeF(){
    box.style.fontSize=16+'px';
}
function decreaseSizeF(){
    size-=2;
    box.style.fontSize=size+'px';
}
function increaseSizeF(){
    size+=2;
    box.style.fontSize=size+'px';
}
